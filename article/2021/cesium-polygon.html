window.entity = new Cesium.Entity({
  polygon: new Cesium.PolygonGraphics({
    hierarchy: buildPolygonHierarchy(polygonJson.geometry.coordinates)
  })
})

function buildPolygonHierarchy(coords, index) {
  index === undefined && (index = 0)
  return new Cesium.PolygonHierarchy(
    Cesium.Cartesian3.fromDegreesArrayHeights(
      coords[index]
        .map((i) => {
          return i.length === 2 ? i.concat(0) : i
        })
        .flat()
    ),
    index === coords.length - 1
      ? []
      : [buildPolygonHierarchy(coords, index + 1)]
  )
}

viewer.entities.add(entity)
